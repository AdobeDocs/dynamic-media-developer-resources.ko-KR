  <td class="stentry"> <p><span class="codeph"> <span class="varname"> py, py</span> </span> </p></td> 
 </tr> 
 <tr class="strow"> 
  <td class="stentry"> <p><span class="codeph"> <span class="varname"> py, py</span> </span> </p></td> 
  <td class="stentry"> <p>x, y 값(픽셀 단위)(int, int) </p></td> 
 </tr>
